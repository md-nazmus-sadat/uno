!UNO!

#include stdio.h

#include stdbool.h

int STACK[5];

int TOP=-1;

void push(){

if (TOP==4)

printf("Stack Overflow\n");

else{

int n;

printf("Enter a no. : ");

scanf("%d",&n);

TOP++;

STACK[TOP]=n;

}

}

void pop(){

if (TOP==-1)

printf("Stack Underflow\n");

else{

int temp=STACK[TOP];

printf("The popped element is : %d\n",temp);

TOP--;

}

}

void peek(){

printf("%d\n",STACK[TOP]);

}

void display(){

int i;

for (i=TOP;i>=0;i--)

printf("%d\n",STACK[i]);

}

int main(){

while(true){

printf("Press 1 to push an element in the stack\n");

printf("Press 2 to pop an element from the stack\n");

printf("Press 3 to peek an element from the stack\n");

printf("Press 4 to display the elements of the stack\n");

printf("Press 5 to exit\n");

printf("Enter your choice:

int n;

scanf("%d",&n);

switch(n){

case 1:push(); break;

case 2:pop(); break;

case 3:peek(); break;

case 4:display(); break;

case 5: return;

default:printf("Invalid Choice\n");

}

}

return 0;

}
